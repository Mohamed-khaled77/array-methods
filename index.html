4.push( )

هي دالة تُستخدم لإضافة عنصر أو أكثر إلى نهاية المصفوفة تقوم بتعديل المصفوفة الأصلية من خلال زيادة طولها بمقدار العناصر المضافة، وتُرجع الطول الجديد للمصفوفة.

5.shift( )

هي دالة تُستخدم لإزالة العنصر الأول من المصفوفة وإرجاعه. تقوم هذه لدالة بتعديل المصفوفة الأصلية عن طريق تقليل طولها بمقدار عنصر واحد، كما يتم نقل العناصر المتبقية إلى اليسار (أو إلى بداية المصفوفة) لملء الفراغ الذي تركه العنصر المُزال

exsampel

let fruits = ['Apple', 'Banana', 'Cherry'];
let firstFruit = fruits.shift ();
console. log(firstFruit); // "Apple,Banana,Cherry"
console. log(fruits); // "Apple,Banana,Cherry"

10.splice( )

هي دالة تقوم بتعديل المصفوفة الاصلية مباشرة وتعيد مصفوفة جديدة يتم بها حذف او استبدال العناصر

exsampel

fruits.splice (2, 0, "cherry");
console. log(fruits); // ["apple", "banana", "cherry", "date"]

11.slice( )

دالة تستخدم لاستخراج جزء من مصفوفة او سلسلة نصية بدون تعديل في المصفوفة او السلسلة الاصلية وتقوم بارجاع مصفوفة جديدة تحتوي على العناصر المستخرجة

exsampel

let fruits = ["apple", "banana", "cherry", "date", "elderberry"];
let slicedFruits = fruits.slice(1, 3);
console.log(slicedFruits); // ["banana", "cherry"]
console.log(fruits); // ["apple", "banana", "cherry", "date", "elderberry"]

12.reverse( )

numbers.reverse();
console. log(numbers); // [5, 4, 3, 2, 1]

14.indexof( )

ستخدم للعثور على الفهرس الأول لعنصر معين داخل المصفوفة أو السلسلة النصية. إذا تم العثور على العنصر، تُرجع الدالة فهرسه (الذي يبدأ من 0). إذا لم يتم العثور على العنصر، تُرجع -1.

exsampel

let fruits = ["apple", "banana", "cherry", "apple"];
let index =fruits.indexOf("banana");
console.log(index); // 1

16.find( )

دالة تستخدم للعثور على اول عنصر في المصفوفة يطابق شرطا معينا يتم تحدية بواسطة دالة اختبار بمجرد العصور على العنصر تقوم الدالة بارجاعة اما اذا لم يتم العثور علية ترجع undefined

exsampel

let numbers = [10, 20, 30, 40, 50];
let result = numbers.find(element => element > 25);
console.log(result); // 30

exsampel

let numbers = [10, 20, 30, 40, 50];
let result = numbers.findIndex(element => element > 25);
console.log(result); // 2

20.every( )

دالة تستخدم لاختبار ما اذا كانت جميع العناصر في المصفوفة تطابق شرطا معينا يحدده المستخدم في دالة اختبار اذا كان الشرط ينطبق على جميع العناصر تقوم بارجاع ترو واذا وجد عنصر واحد لا ينطبق علية الشرط ترجع فولس

exsampel

let numbers = [10, 20, 30, 40];
let allGreaterThan5 = numbers.every(num => num > 5);
console.log(allGreaterThan5); // true

دالة تستخدم لاختبار ما اذا كان اي عنصر في المصفوفة يطابق شرطا ادخلة المستخدم في دالة اختبار اذا كان هناك اي عنصر يطابق الشرط ترجع ترو واذا لم يوجد ترجع فولس

exsampel

let numbers = [10, 20, 30, 40];
let hasGreaterThan25 = numbers.some(num => num > 25);
console.log(hasGreaterThan25); // true

22.fill( )

دالة تُستخدم لملء جميع عناصر المصفوفة بقيمة ثابتة، بدءًا من فهرس محدد (اختياري) إلى فهرس نهائي (اختياري). هذه الدالة تعدل المصفوفة الأصلية

exsampel

let numbers = [1, 2, 3, 4, 5];
numbers.fill(0);
console.log(numbers); // [0, 0, 0, 0, 0]

23.copywithin( )

دالة تُستخدم لنسخ جزء من المصفوفة إلى موقع آخر داخل نفس المصفوفة دون تغيير حجمها. يتم النسخ على مستوى المصفوفة الأصلية، أي أن الدالة تعدل المصفوفة الأصلية.

exsampel

let arr = [1, 2, 3, 4, 5];
arr.copyWithin(0, 3);
console.log(arr); // [4, 5, 3, 4, 5]

25.foreach( )

26.map( )

دالة تستخدم لانشاء مصفوفة جديدة تحتوي على نتائج تطبيق دالة معينة على كل عنصر من عناصر المصفوفة الاصلية تعتبر مفيدة عندما تحتاج الى تحويل كل عنصر في المصفوفة الي قيمة جديدة بناء عملية معينة وتقوم بارجاع مصفوفة جديدة ولا تقوم بتعديل المصفوفة الاصلية

exsampel

let numbers =[1, 2, 3, 4];
let squaredNumbers = numbers. map(function(number) {
return number * number; }) console. log(squaredNumbers); // [1, 4, 9, 16]

letevenNumbers = numbers. filter(function(number){

28.reduce( )

دالة تستخدم لتطبيق عملية جمع داخل دالة وتقوم بجمع كل عنصر في المصفوفة وتقوم بارجاع النتيجة فقط

exsampel

let numbers = [1, 2, 3, 4];
let sum = numbers.reduce(function(accumulator, currentValue){
return accumulator + currentValue; },0); console.log(sum); // 10

29.reduceright( )

دالة تقوم بجمع عنتصر المصفوفة ولاكن تقوم بالعملية من اليمين الى الياسار ويقووم بتطبيق التجميع على كل عنصر في المصفوفة

exsampel

let numbers =[1, 2, 3, 4];
let result = numbers.reduceRight(function(accumulator, currentValue){
return accumulator - currentValue; }, 0); console.log(result); // -4
